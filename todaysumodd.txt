import java.util.Scanner;
 
class Example1 {
 
  public static void main(String args[]) {
    Scanner s = new Scanner(System.in);
    int num = 0, sum = 0;
    System.out.println("Enter num value");
    num = s.nextInt(); // 100
 
    for (int i = 1; i <= num; i++) {
      if (i % 2 != 0) {
        sum = sum + i;
      }
    }
    System.out.println("sum of odd numbers : " + sum);
 
  }
 
}
